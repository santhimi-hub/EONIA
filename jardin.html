export default { async fetch(request, env) { const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type, Authorization, Notion-Version", }; if (request.method === "OPTIONS") return new Response(null, { headers: corsHeaders }); const TOKEN = env.NOTION_KEY; const DB_PLAQUES_ID = env.NOTION_DB_PLAQUES_ID; try { let emailClient = null; let plaqueIdRecherche = null; if (request.method === "POST") { const reqBody = await request.json().catch(() => ({})); emailClient = reqBody.email ? reqBody.email.toLowerCase().trim() : null; const url = new URL(request.url); plaqueIdRecherche = reqBody.id || url.searchParams.get("id"); } if (!plaqueIdRecherche) return new Response(JSON.stringify({ error: "Pas d'ID" }), { status: 400, headers: corsHeaders }); // 1. RECHERCHE DE LA PLAQUE (On utilise "title" par sécurité car c'est la colonne principale) const plaqueResponse = await fetch(`https://api.notion.com/v1/databases/${DB_PLAQUES_ID}/query`, { method: "POST", headers: { "Authorization": `Bearer ${TOKEN}`, "Notion-Version": "2022-06-28", "Content-Type": "application/json", }, body: JSON.stringify({ filter: { property: "title", title: { equals: plaqueIdRecherche.trim() } } }), }); const dataPlaque = await plaqueResponse.json(); const plaquePage = dataPlaque.results?.[0]; if (!plaquePage) return new Response(JSON.stringify({ error: "Plaque introuvable" }), { status: 404, headers: corsHeaders }); const p = plaquePage.properties; const rowEmail = p["Propriétaire"]?.email?.toLowerCase().trim(); const isOwner = emailClient && rowEmail === emailClient; // 2. RÉCUPÉRATION DU CATALOGUE (Lien_Plante) let bot = {}; const relation = p["Lien_Plante"]?.relation; if (relation && relation.length > 0) { const botanicResponse = await fetch(`https://api.notion.com/v1/pages/${relation[0].id}`, { headers: { "Authorization": `Bearer ${TOKEN}`, "Notion-Version": "2022-06-28" } }); const botData = await botanicResponse.json(); bot = botData.properties || {}; } const getText = (prop) => { if (!prop) return ""; const array = prop.title || prop.rich_text || []; return array.length > 0 ? array[0].plain_text : ""; }; // 3. RÉPONSE (On garde tes clés d'origine) const result = { "pageId": plaquePage.id, "ID_Plaque": plaqueIdRecherche, "Surnom de ma plante": getText(p["Surnom_Perso"]), "Notes_Suivi": getText(p["Note_Privee"]), "Latitude": p["Lat"]?.number || null, "Longitude": p["Long"]?.number || null, "estProprietaire": isOwner, // Retour aux noms qui fonctionnaient "Nom commun": getText(bot["Nom commun"]) || "Spécimen Éonia", "Famille": bot["Famille"]?.select?.name || "", "Exposition": bot["Exposition"]?.select?.name || "", "Arrosage": bot["Arrosage"]?.select?.name || "", "Rusticité": getText(bot["Rusticité"]), "Taille maturité": getText(bot["Taille maturité"]), "Photo": bot["Photo"]?.files?.[0]?.file?.url || bot["Photo"]?.files?.[0]?.external?.url || "", "Type": bot["Type"]?.multi_select?.map(s => s.name) || [], "Entretien": getText(bot["Entretien"]), "Protection et sante": getText(bot["Protection et sante"]), "Campagnonnage / Plantes associees": getText(bot["Campagnonnage / Plantes associees"]) }; return new Response(JSON.stringify([result]), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } catch (err) { return new Response(JSON.stringify({ error: err.message }), { status: 500, headers: corsHeaders }); } } };
